15// NOTE: Standard UdpSocket could not enable SO_REUSEADDR
16// use nix::sys::socket::sockopt::{IpMulticastLoop, ReuseAddr, ReusePort};
17use crate::transport::PORT;
18use crate::transport::error::{BindError, ScoketError};
19use crate::transport::result::Result;
20#[cfg(feature = "unix")]
21use log::warn;
22#[cfg(feature = "unix")]
23use nix::sys::socket::{Shutdown, shutdown};
24use std::io;
25use std::net::{Ipv4Addr, Ipv6Addr, SocketAddr};
26#[cfg(feature = "unix")]
27use std::os::unix::io::AsRawFd;
28use std::{thread, time};
29
30const UDP_SOCKET_BIND_RETRY_MAX: u32 = 3;
31const UDP_SOCKET_BIND_SLEEP_MSEC: u64 = 100;
32
33pub struct UdpSocket {
34    sock: Option<std::net::UdpSocket>,
35    ifaddr: Option<SocketAddr>,
36}
37
38#[cfg(feature = "unix")]
39fn create_socket_v4(ifaddr: SocketAddr) -> io::Result<std::net::UdpSocket> {
40    net2::UdpBuilder::new_v4()?
41        .reuse_address(true)?
42        // .reuse_port(true)?
43        .bind(ifaddr)
44}
45
46#[cfg(not(feature = "unix"))]
47fn create_socket_v4(ifaddr: SocketAddr) -> io::Result<std::net::UdpSocket> {
48    let sock = std::net::UdpSocket::bind(ifaddr.to_string());
49    sock
50}
51
52#[cfg(feature = "unix")]
53fn create_socket_v6(ifaddr: SocketAddr) -> io::Result<std::net::UdpSocket> {
54    net2::UdpBuilder::new_v6()?
55        .reuse_address(true)?
56        // .reuse_port(true)?
57        .bind(ifaddr)
58}
59
60#[cfg(not(feature = "unix"))]
61fn create_socket_v6(ifaddr: SocketAddr) -> io::Result<std::net::UdpSocket> {
62    let sock = std::net::UdpSocket::bind(ifaddr.to_string());
63    sock
64}
65
66impl UdpSocket {
67    pub fn new() -> UdpSocket {
68        UdpSocket {
69            sock: None,
70            ifaddr: None,
71        }
72    }
73
74    pub fn local_addr(&self) -> io::Result<SocketAddr> {
75        if self.ifaddr.is_some() {
76            return Ok(self.ifaddr.unwrap());
77        }
78        Err(std::io::Error::new(
79            std::io::ErrorKind::Other,
80            "socket is not bound",
81        ))
82    }
83
84    pub fn bind(&mut self, ifaddr: SocketAddr) -> Result<()> {
85        if self.sock.is_some() {
86            self.close();
87        }
88
89        let mut sock: Option<std::net::UdpSocket> = None;
90
91        for _n in 0..UDP_SOCKET_BIND_RETRY_MAX {
92            if ifaddr.is_ipv4() {
93                let sock_v4 = create_socket_v4(ifaddr);
94                if sock_v4.is_ok() {
95                    sock = Some(sock_v4.unwrap());
96                    break;
97                }
98            } else if ifaddr.is_ipv6() {
99                let sock_v6 = create_socket_v6(ifaddr);
100                if sock_v6.is_ok() {
101                    sock = Some(sock_v6.unwrap());
102                    break;
103                }
104            }
105            thread::sleep(time::Duration::from_millis(UDP_SOCKET_BIND_SLEEP_MSEC));
106        }
107
108        if sock.is_none() {
109            return Err(ScoketError::new(&format!("could not bind to {}", ifaddr)));
110        }
111
112        let _ = sock
113            .as_ref()
114            .unwrap()
115            .set_read_timeout(Some(time::Duration::from_millis(500)));
116        self.sock = sock;
117        self.ifaddr = Some(ifaddr);
118        Ok(())
119    }
120
121    pub fn close(&self) -> bool {
122        if self.sock.is_none() {
123            return true;
124        }
125        #[cfg(feature = "unix")]
126        {
127            let fd = self.sock.as_ref().unwrap().as_raw_fd();
128            let res = shutdown(fd, Shutdown::Both);
129            if res.is_err() {
130                warn!("shutdown ({})", res.err().unwrap().desc());
131            }
132        }
133        true
134    }
135
136    pub fn send_to(&self, buf: &[u8], to_addr: SocketAddr) -> Result<usize> {
137        if self.sock.is_none() {
138            return Err(BindError::new());
139        }
140        let res = self.sock.as_ref().unwrap().send_to(buf, to_addr);
141        if res.is_err() {
142            return Err(ScoketError::new(&format!("{:?}", res.err())));
143        }
144        Ok(res.unwrap())
145    }
146
147    pub fn recv_from(&self, buf: &mut [u8]) -> Result<(usize, SocketAddr)> {
148        if self.sock.is_none() {
149            return Err(BindError::new());
150        }
151        self.sock.as_ref().unwrap().recv_from(buf)
152    }
153
154    pub fn join_multicast_v4(&mut self, multiaddr: &Ipv4Addr, ifaddr: &Ipv4Addr) -> Result<()> {
155        if self.sock.is_none() {
156            return Err(BindError::new());
157        }
158        self.ifaddr = Some(format!("{}:{}", ifaddr, PORT).parse().unwrap());
159        self.sock
160            .as_ref()
161            .unwrap()
162            .join_multicast_v4(multiaddr, ifaddr)
163    }
164
165    pub fn join_multicast_v6(&mut self, multiaddr: &Ipv6Addr, ifaddr: &Ipv6Addr) -> Result<()> {
166        if self.sock.is_none() {
167            return Err(BindError::new());
168        }
169        self.ifaddr = Some(format!("{}:{}", ifaddr, PORT).parse().unwrap());
170        self.sock.as_ref().unwrap().join_multicast_v6(multiaddr, 0)
171    }
172}
173
174impl Drop for UdpSocket {
175    fn drop(&mut self) {
176        self.close();
177    }
178}
